#include <avr/io.h>
#include <util/delay.h>
#include <avr/interrupt.h>

int state = 0;
unsigned char bright[4] = {0, 80, 160, 255};

ISR(INT0_vect)
{
    cli();
    _delay_ms(50);

    state++;
    if (state > 3)
    {
        state = 0;
    }

    OCR0 = bright[state];

    sei();
    EIFR = (1 << INTF0);
}

main()
{
    SREG |= 0x80;

    DDRB |= (1 << DDB4);

    DDRD &= ~(1 << DDD0);
    PORTD |= (1 << PD0);

    EICRA = (1 << ISC01);
    EIMSK |= (1 << INT0);

    TCCR0 = (1 << WGM01) | (1 << WGM00) | (1 << COM01) | (1 << CS01) | (1 << CS00);

    OCR0 = bright[state];

    sei();

    while (1)
    {
    }
}
